1、WITH xxx AS的使用

    WITH WDA AS(
    SELECT * FROM CS_USER
    WHERE USER_PHONE='15169160503'
    )
    //上面创建wda临时表，下面的操作都在wda里操作
    SELECT A.*,B.* FROM
    (
    SELECT COUNT(*) USER_COUNT1 FROM WDA
    WHERE NICK_NAME='大神'
    ) A LEFT JOIN (
    SELECT COUNT(*) USER_COUNT2 FROM WDA
    WHERE USER_PHONE='15169160503'
    ) B ON 1=1
