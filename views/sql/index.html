1、WITH xxx AS的使用

    WITH WDA AS(
    SELECT * FROM CS_USER
    WHERE USER_PHONE='15169160503'
    )
    //上面创建wda临时表，下面的操作都在wda里操作
    SELECT A.*,B.* FROM
    (
    SELECT COUNT(*) USER_COUNT1 FROM WDA
    WHERE NICK_NAME='大神'
    ) A LEFT JOIN (
    SELECT COUNT(*) USER_COUNT2 FROM WDA
    WHERE USER_PHONE='15169160503'
    ) B ON 1=1

2、CASE WHEN THEN ELSE END的使用

    select user_name, (
	case when user_sex='01' then '男'
	when user_sex='02' then '女'
	else '保密' end
    ) user_sex
    from t_sys_user

    ==========================================

    select user_name,(
	case user_sex
	when '01' then '男'
	when '02' then '女'
	else '保密' end
    ) user_sex
    from t_sys_user

    以上是2种方法
